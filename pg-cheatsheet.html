查看表结构 & 数据字典

PostgreSQL 中查看表、字段、索引、约束等元信息的常用方法。

\d 表名
psql 元命令，查看表结构（最常用）
\d emp \d+ emp -- 含描述、存储大小 \d emp_id_pk -- 查看索引/约束详情
仅在 psql 命令行中有效
information_schema.columns
标准 SQL 方式，任何客户端都能用
SELECT column_name, data_type, is_nullable, column_default FROM information_schema.columns WHERE table_name = 'emp' ORDER BY ordinal_position;
pg_catalog 查字段详情
PostgreSQL 原生方式，信息最全
SELECT a.attname AS col, pg_catalog.format_type(a.atttypid, a.atttypmod) AS type, NOT a.attnotnull AS nullable FROM pg_catalog.pg_attribute a WHERE a.attrelid = 'emp'::regclass AND a.attnum > 0 AND NOT a.attisdropped ORDER BY a.attnum;
查看所有表
列出当前 schema 下的所有表
-- psql 元命令 \dt \dt+ -- 含大小/描述 -- SQL 方式 SELECT table_name FROM information_schema.tables WHERE table_schema = 'public';
查看索引
查看表的索引及索引列
-- psql \d emp -- SQL SELECT indexname, indexdef FROM pg_indexes WHERE tablename = 'emp';
查看约束
主键 / 外键 / 唯一 / CHECK 约束
SELECT conname, contype, pg_get_constraintdef(oid) FROM pg_constraint WHERE conrelid = 'emp'::regclass; -- psql 快捷 \d emp
查看视图
列出视图及其定义
-- psql \dv \dv+ -- 含描述 -- SQL 查视图定义 SELECT viewname, definition FROM pg_views WHERE schemaname = 'public';
查看函数 / 存储过程
列出函数和过程源码
-- psql \df \df+ -- 含源码 -- SQL SELECT routine_name, routine_type FROM information_schema.routines WHERE routine_schema = 'public';

SELECT 查询语句

PostgreSQL 数据查询完整语法，含聚合、连接、窗口函数、CTE 等。

基础查询
SELECT / FROM / WHERE / LIMIT
SELECT id, name, dept FROM emp WHERE dept = 'IT' AND sal > 5000 LIMIT 10;
分页（LIMIT / OFFSET）
PostgreSQL 标准分页方式
-- 第 3 页，每页 10 条 SELECT * FROM emp ORDER BY id LIMIT 10 OFFSET 20; -- 等价于 SELECT * FROM emp ORDER BY id LIMIT 20, 10; -- MySQL 风格，不推荐
聚合 & GROUP BY
分组统计，HAVING 过滤分组
SELECT dept, COUNT(*) AS cnt, AVG(sal) AS avg_sal, MAX(sal) AS max_sal FROM emp GROUP BY dept HAVING COUNT(*) > 3;
排序
ORDER BY / ASC / DESC / NULLS
SELECT * FROM emp ORDER BY dept ASC, sal DESC NULLS LAST;
连接查询
INNER / LEFT / RIGHT / FULL JOIN
SELECT e.name, d.dept_name FROM emp e LEFT JOIN dept d ON e.dept_id = d.id; -- USING 简写（列名相同时） SELECT e.name, d.dept_name FROM emp e LEFT JOIN dept d USING (dept_id);
子查询 & EXISTS
子查询、IN、EXISTS
SELECT name FROM emp WHERE dept_id IN ( SELECT id FROM dept WHERE loc = '深圳' ); SELECT name FROM emp e WHERE EXISTS ( SELECT 1 FROM orders o WHERE o.emp_id = e.id );
CTE（WITH 子句）
公共表表达式，可读性强，可递归
WITH high_sal AS ( SELECT * FROM emp WHERE sal > 10000 ) SELECT dept, COUNT(*) FROM high_sal GROUP BY dept; -- 递归 CTE WITH RECURSIVE t(n) AS ( SELECT 1 UNION ALL SELECT n+1 FROM t WHERE n < 100 ) SELECT * FROM t;
窗口函数
ROW_NUMBER / RANK / DENSE_RANK / LAG / LEAD
SELECT name, dept, sal, RANK() OVER( PARTITION BY dept ORDER BY sal DESC ) AS rank_in_dept, LAG(sal) OVER(ORDER BY id) AS prev_sal FROM emp;
集合运算
UNION / INTERSECT / EXCEPT
SELECT name FROM emp UNION -- 去重 SELECT name FROM contractor; SELECT dept_id FROM emp INTERSECT -- 交集 SELECT id FROM dept; SELECT id FROM dept EXCEPT -- 差集（PG 用 EXCEPT，不用 MINUS） SELECT dept_id FROM emp;
常用函数
COALESCE / CASE / 字符串 / 日期
COALESCE(comm, 0) -- NULL 替换 CASE WHEN ... END -- 条件 LEFT(name,3) / SUBSTRING(...) -- 截取 LENGTH(name) -- 长度 NOW() / CURRENT_TIMESTAMP -- 当前时间 DATE_TRUNC('month', now()) -- 截断日期 EXTRACT(YEAR FROM hiredate) -- 提取日期部分
DISTINCT / FILTER
去重及聚合中的条件过滤
SELECT DISTINCT dept FROM emp; -- PG 特有：聚合 FILTER SELECT COUNT(*) FILTER (WHERE sal > 10000) AS high_cnt, COUNT(*) FILTER (WHERE sal <= 10000) AS low_cnt FROM emp;
GROUPING SETS / CUBE / ROLLUP
多维聚合（PG 特有支持）
-- 同时按 dept 和 全表 聚合 SELECT dept, COUNT(*) FROM emp GROUP BY GROUPING SETS (dept, ()); -- ROLLUP：层次聚合 SELECT dept, job, COUNT(*) FROM emp GROUP BY ROLLUP(dept, job);

DML（增删改）

INSERT / UPDATE / DELETE / UPSERT（ON CONFLICT）。

INSERT
插入单行 / 多行
INSERT INTO emp (id, name, dept) VALUES (1, 'Alice', 'IT'); -- 多行插入 INSERT INTO emp (id, name) VALUES (2, 'Bob'), (3, 'Cathy'), (4, 'David');
INSERT ... RETURNING
PG 特色：插入后直接返回结果（无需再查）
INSERT INTO emp (name, dept) VALUES ('Alice', 'IT') RETURNING id, name; -- 返回自增 ID INSERT INTO emp (name) VALUES ('Bob') RETURNING id;
UPDATE
更新数据，支持 FROM 关联更新
UPDATE emp SET sal = sal * 1.1, updated_at = NOW() WHERE dept = 'IT'; -- 关联更新 UPDATE emp e SET dept_name = d.dept_name FROM dept d WHERE e.dept_id = d.id;
UPDATE ... RETURNING
更新后直接返回受影响的行
UPDATE emp SET sal = sal * 1.1 WHERE dept = 'IT' RETURNING id, name, sal;
DELETE
删除数据
DELETE FROM emp WHERE id = 1; -- 清空表（可回滚，DML） DELETE FROM emp; -- 截断（DDL，不可回滚，速度快） TRUNCATE TABLE emp; TRUNCATE TABLE emp RESTART IDENTITY; -- 重置自增
DELETE ... RETURNING
删除后返回被删除的行
DELETE FROM emp WHERE dept = 'TEMP' RETURNING id, name;
ON CONFLICT（UPSERT）
PG 特有：有则更新，无则插入
INSERT INTO emp (id, name, sal) VALUES (1, 'Alice', 8000) ON CONFLICT (id) DO UPDATE SET name = EXCLUDED.name, sal = EXCLUDED.sal; -- 冲突时什么都不做 ON CONFLICT (id) DO NOTHING
COPY（批量导入导出）
PG 高性能批量数据交换
-- 导出到文件 COPY emp TO '/tmp/emp.csv' CSV HEADER; -- 从文件导入 COPY emp FROM '/tmp/emp.csv' CSV HEADER; -- 导出查询结果 COPY (SELECT * FROM emp WHERE dept='IT') TO '/tmp/it_emp.csv' CSV;
需要超级用户权限，或用 \copy（psql）

DDL（建表 / 改表 / 删表）

PostgreSQL 数据定义语句，含序列、约束、枚举等特有类型。

CREATE TABLE
创建表，含主键、默认值、注释
CREATE TABLE emp ( id SERIAL PRIMARY KEY, name VARCHAR(50) NOT NULL, dept VARCHAR(20) DEFAULT 'IT', sal NUMERIC(10,2), hiredate DATE DEFAULT CURRENT_DATE ); COMMENT ON TABLE emp IS '员工表'; COMMENT ON COLUMN emp.name IS '姓名';
ALTER TABLE
修改表结构
-- 加列 ALTER TABLE emp ADD COLUMN email VARCHAR(100); -- 改列类型 ALTER TABLE emp ALTER COLUMN name TYPE VARCHAR(100); -- 改列名 ALTER TABLE emp RENAME COLUMN sal TO salary; -- 改表名 ALTER TABLE emp RENAME TO employees; -- 删列 ALTER TABLE emp DROP COLUMN email;
DROP TABLE
删除表
DROP TABLE emp; DROP TABLE emp CASCADE; -- 级联删除依赖对象 DROP TABLE IF EXISTS emp; -- 不报错 DROP TABLE emp CASCADE; -- 同时删外键引用
SERIAL / GENERATED
自增主键（PG 特有）
-- 旧方式（仍广泛用） id SERIAL PRIMARY KEY -- 新方式（PG 10+ 推荐） id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY -- 获取当前值 SELECT currval('emp_id_seq'); SELECT lastval();
CREATE INDEX
创建索引（含 PG 特有索引类型）
CREATE INDEX idx_emp_name ON emp(name); CREATE UNIQUE INDEX uq_emp_email ON emp(email); -- PG 特有：部分索引 CREATE INDEX idx_active ON emp(dept) WHERE status='active'; -- GIN 索引（JSONB / 数组） CREATE INDEX idx_data ON emp USING GIN (data_json);
约束
主键 / 外键 / 唯一 / CHECK
CREATE TABLE emp ( id INTEGER PRIMARY KEY, dept_id INTEGER REFERENCES dept(id), email VARCHAR(100) UNIQUE, sal NUMERIC CHECK (sal > 0) ); -- 添加约束 ALTER TABLE emp ADD CONSTRAINT fk_dept FOREIGN KEY (dept_id) REFERENCES dept(id);
CREATE VIEW
创建视图
CREATE OR REPLACE VIEW v_emp AS SELECT id, name, dept FROM emp WHERE dept = 'IT'; -- 可更新视图（满足条件时） CREATE VIEW v_active AS SELECT * FROM emp WHERE status='active';
枚举 & 域类型
PG 特有：ENUM 和 DOMAIN
-- 枚举类型 CREATE TYPE dept_enum AS ENUM ('IT','HR','Finance'); ALTER TABLE emp ALTER COLUMN dept TYPE dept_enum USING dept::dept_enum; -- 域（带约束的类型） CREATE DOMAIN email_domain AS VARCHAR(255) CHECK (VALUE ~ '^[^@]+@[^@]+\\.[^@]+$');

事务 & 锁

PostgreSQL 事务控制及锁机制。

基本事务
BEGIN / COMMIT / ROLLBACK
BEGIN; INSERT INTO emp (name) VALUES ('Alice'); UPDATE dept SET cnt = cnt + 1; COMMIT; -- 或 BEGIN; ... ROLLBACK;
SAVEPOINT
保存点，部分回滚
BEGIN; INSERT INTO emp (name) VALUES ('Alice'); SAVEPOINT sp1; INSERT INTO emp (name) VALUES ('Bob'); ROLLBACK TO sp1; -- 只撤销 Bob 的插入 COMMIT;
事务隔离级别
设置事务隔离级别
BEGIN ISOLATION LEVEL REPEATABLE READ; ... COMMIT; -- 或 SET TRANSACTION ISOLATION LEVEL SERIALIZABLE; -- 查看当前级别 SHOW TRANSACTION ISOLATION LEVEL;
FOR UPDATE（行锁）
查询时锁定行，防止并发修改
BEGIN; SELECT * FROM emp WHERE id = 1 FOR UPDATE; UPDATE emp SET sal = sal + 1000 WHERE id = 1; COMMIT; -- 跳过已锁定的行 FOR UPDATE SKIP LOCKED -- 等待锁释放 FOR UPDATE NOWAIT
咨询锁（Advisory Lock）
PG 特有：应用层自定义锁
-- 获取会话级锁（阻塞其他会话） SELECT pg_advisory_lock(12345); -- 尝试获取（不阻塞） SELECT pg_try_advisory_lock(12345); -- 释放 SELECT pg_advisory_unlock(12345); -- 事务级锁（事务结束自动释放） SELECT pg_advisory_xact_lock(12345);
查看锁等待
诊断锁等待问题
SELECT a.pid, a.usename, a.query, l.locktype, l.mode, a.wait_event_type, a.wait_event FROM pg_stat_activity a JOIN pg_locks l ON a.pid = l.pid WHERE l.granted = false;

PL/pgSQL 函数 & 存储过程

PostgreSQL 的过程语言，用于编写函数（FUNCTION）和存储过程（PROCEDURE）。

CREATE FUNCTION
创建函数（有返回值）
CREATE OR REPLACE FUNCTION get_emp_name(p_id INTEGER) RETURNS VARCHAR LANGUAGE plpgsql AS $$ DECLARE v_name VARCHAR; BEGIN SELECT name INTO v_name FROM emp WHERE id = p_id; RETURN v_name; END; $$;
CREATE PROCEDURE
创建存储过程（PG 11+，无返回值，支持事务）
CREATE OR REPLACE PROCEDURE add_emp(p_name VARCHAR, p_dept VARCHAR) LANGUAGE plpgsql AS $$ BEGIN INSERT INTO emp (name, dept) VALUES (p_name, p_dept); COMMIT; -- 过程内可提交 END; $$; -- 调用（必须用 CALL） CALL add_emp('Alice', 'IT');
返回多行（SETOF / TABLE）
函数返回查询结果集
CREATE FUNCTION get_emps_by_dept(p_dept VARCHAR) RETURNS TABLE(id INTEGER, name VARCHAR) LANGUAGE plpgsql AS $$ BEGIN RETURN QUERY SELECT e.id, e.name FROM emp e WHERE e.dept = p_dept; END; $$; SELECT * FROM get_emps_by_dept('IT');
控制结构
IF / CASE / LOOP / WHILE / FOR
-- IF IF v_sal > 10000 THEN ... ELSIF v_sal > 5000 THEN ... ELSE ... END IF; -- CASE CASE v_dept WHEN 'IT' THEN v_bonus := 1000; ELSE v_bonus := 500; END CASE; -- FOR 循环（遍历查询结果） FOR rec IN SELECT * FROM emp LOOP RAISE NOTICE 'name=%', rec.name; END LOOP;
异常处理
EXCEPTION 块
BEGIN ... EXCEPTION WHEN NO_DATA_FOUND THEN RAISE NOTICE '未找到数据'; WHEN OTHERS THEN RAISE EXCEPTION '错误: %', SQLERRM; END; -- 捕获特定错误码 EXCEPTION WHEN unique_violation THEN ...
RAISE（日志输出）
PL/pgSQL 中的调试输出
RAISE NOTICE 'x = %', x; RAISE WARNING '注意: %', msg; RAISE EXCEPTION '出错: %', err_msg; -- 抛出异常 -- 日志级别：DEBUG / LOG / INFO / NOTICE / WARNING / EXCEPTION

完整 PL/pgSQL 函数模板

CREATE OR REPLACE FUNCTION update_salary(
  p_id     -- 参数（无需指定 IN/OUT，默认 IN）
  p_pct    NUMERIC
)
RETURNS NUMERIC
LANGUAGE plpgsql
AS $$
-- 变量声明
DECLARE
  v_old_sal NUMERIC;
  v_new_sal NUMERIC;

BEGIN
  -- 查询当前薪资
  SELECT sal INTO v_old_sal
  FROM emp WHERE id = p_id;

  -- 计算新薪资
  v_new_sal := v_old_sal * (1 + p_pct / 100.0);

  -- 更新
  UPDATE emp
  SET sal = v_new_sal,
      updated_at = NOW()
  WHERE id = p_id;

  -- 返回新薪资
  RETURN v_new_sal;

-- 异常处理
EXCEPTION
  WHEN NO_DATA_FOUND THEN
    RAISE EXCEPTION '员工不存在: %', p_id;
  WHEN OTHERS THEN
    RAISE EXCEPTION '更新失败: %', SQLERRM;

END;
$$;

索引 & 性能优化

PostgreSQL 索引类型及性能诊断命令。

B-tree 索引（默认）
最常用，适合等值、范围查询
CREATE INDEX idx_emp_name ON emp(name); -- 联合索引 CREATE INDEX idx_emp_dept_sal ON emp(dept, sal);
Hash 索引
等值查询，PG 10+ 支持 WAL 可恢复
CREATE INDEX idx_emp_email_hash ON emp USING HASH (email);
GIN 索引
JSONB / 数组 / 全文搜索
-- JSONB CREATE INDEX idx_data ON emp USING GIN (data_json); -- 数组 CREATE INDEX idx_tags ON emp USING GIN (tags); -- 全文搜索 CREATE INDEX idx_fts ON emp USING GIN (to_tsvector('chinese', content));
GiST / SP-GiST 索引
几何数据、范围类型、模糊搜索
-- 几何 CREATE INDEX idx_geom ON places USING GIST (geom); -- 模糊搜索（pg_trgm） CREATE INDEX idx_name_trgm ON emp USING GIST (name gist_trgm_ops);
部分索引
只对部分行建索引，节省空间
CREATE INDEX idx_active_emp ON emp(dept) WHERE status = 'active';
EXPLAIN / EXPLAIN ANALYZE
查看执行计划（性能调优核心）
-- 只看计划 EXPLAIN SELECT * FROM emp WHERE dept='IT'; -- 实际执行 + 耗时 EXPLAIN ANALYZE SELECT * FROM emp WHERE dept='IT'; -- 更详细 EXPLAIN (ANALYZE, BUFFERS) SELECT * FROM emp WHERE dept='IT';
查看索引使用统计
找出未使用的索引
SELECT schemaname, tablename, indexname, idx_scan, idx_tup_read FROM pg_stat_user_indexes ORDER BY idx_scan;
重建索引
索引膨胀时重建
REINDEX INDEX idx_emp_name; REINDEX TABLE emp; -- 重建表所有索引 REINDEX DATABASE mydb; -- 重建整个库

psql 元命令速查

psql 命令行工具专用斜杠命令（\ 开头），不是 SQL，只在 psql 中有效。

\c / \conninfo
连接数据库
\c mydb -- 切换到 mydb \c mydb user host -- 指定用户/主机 \conninfo -- 显示当前连接信息
\d 系列（查看结构）
最常用的一组元命令
\d -- 所有表/视图/序列 \dt -- 只看表 \dv -- 只看视图 \di -- 只看索引 \ds -- 只看序列 \df -- 只看函数 \d emp -- 查看 emp 表结构
\l / \du / \dg
查看数据库 / 用户 / 角色
\l -- 列出所有数据库 \du -- 列出所有用户/角色 \dg -- 列出所有角色（同 \du） \dn -- 列出所有 schema
\x / \pset
输出格式控制
\x -- 切换扩展显示（行列转置） \x on -- 固定扩展模式 \pset format csv -- CSV 输出 \pset null '(null)' -- NULL 显示方式
\i / \o
执行文件 / 输出重定向
\i /tmp/query.sql -- 执行 SQL 文件 \o /tmp/result.txt -- 输出重定向到文件 \o -- 取消重定向
\copy
psql 客户端 COPY（无需超级用户）
\copy emp TO '/tmp/emp.csv' CSV HEADER \copy emp FROM '/tmp/emp.csv' CSV HEADER
与 SQL COPY 不同，这是 psql 命令
\timing / \echo
计时 / 输出信息
\timing on -- 开启查询计时 \timing off \echo 'Hello' -- 打印信息 \warn '警告' -- 打印到 stderr
\q / \? / \h
退出 / 帮助
\q -- 退出 psql \? -- 元命令帮助 \h CREATE TABLE -- SQL 命令帮助 \h SELECT

  • \d — 列出所有表、视图、序列、外部表
  • \d+ obj — 查看对象结构（含描述、大小）
  • \dt — 只列普通表
  • \dv — 只列视图
  • \di — 只列索引
  • \ds — 只列序列
  • \df — 只列函数
  • \dT — 只列数据类型
  • \dm — 只列物化视图
  • \dS — 只列系统表
