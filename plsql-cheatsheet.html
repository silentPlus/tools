表结构与数据查询（最常用）

日常开发中使用频率最高的命令集，涵盖查表结构、数据字典查询、SELECT 查询及结果处理。

查看表结构

DESC / DESCRIBE
查看表结构、字段类型及是否可为空
DESC emp; DESCRIBE emp;
查询所有表
查看当前用户下所有表
SELECT table_name FROM user_tables; SELECT table_name FROM all_tables WHERE owner = 'SCOTT';
查询表字段
查看表的列名、类型、长度、注释
SELECT column_name, data_type, data_length, nullable FROM user_tab_columns WHERE table_name = 'EMP';
查询字段注释
查看表/列的注释信息
SELECT * FROM user_col_comments WHERE table_name = 'EMP';
查询索引
查看表的索引及索引列
SELECT i.index_name, i.uniqueness, c.column_name FROM user_indexes i JOIN user_ind_columns c ON i.index_name = c.index_name WHERE i.table_name = 'EMP';
查询约束
查看主键、外键、唯一约束等
SELECT constraint_name, constraint_type, status FROM user_constraints WHERE table_name = 'EMP'; -- P=主键 R=外键 U=唯一 C=CHECK
查询存储过程源码
从 user_source 查看过程/函数/包代码
SELECT text FROM user_source WHERE name = 'PKG_EMP' ORDER BY line;
查询表空间占用
查看表的行数和占用空间
SELECT num_rows, blocks, avg_row_len FROM user_tables WHERE table_name = 'EMP'; -- 需先收集统计信息： -- EXEC DBMS_STATS.GATHER_TABLE_STATS('SCOTT','EMP');

SELECT 查询语句

基础查询
SELECT / FROM / WHERE
SELECT id, name, dept FROM emp WHERE dept = 'IT' AND sal > 5000;
聚合查询
GROUP BY / HAVING / 聚合函数
SELECT dept, COUNT(*), AVG(sal) FROM emp GROUP BY dept HAVING COUNT(*) > 3;
排序
ORDER BY / ASC / DESC / NULLS FIRST
SELECT * FROM emp ORDER BY dept ASC, sal DESC NULLS LAST;
连接查询
INNER JOIN / LEFT JOIN / RIGHT JOIN
SELECT e.name, d.dept_name FROM emp e LEFT JOIN dept d ON e.dept_id = d.id;
子查询
标量 / IN / EXISTS 子查询
SELECT name FROM emp WHERE dept_id IN ( SELECT id FROM dept WHERE loc = '深圳' ); SELECT e.name FROM emp e WHERE EXISTS ( SELECT 1 FROM orders o WHERE o.emp_id = e.id );
分页查询（Oracle 12c+）
OFFSET / FETCH 标准分页
SELECT * FROM emp ORDER BY id OFFSET 20 ROWS FETCH NEXT 10 ROWS ONLY;
分页查询（传统）
ROWNUM 嵌套 / ROW_NUMBER()
SELECT * FROM ( SELECT e.*, ROWNUM rn FROM emp e WHERE ROWNUM <= 30 ) WHERE rn > 20; -- 推荐方式（支持排序） SELECT * FROM ( SELECT e.*, ROW_NUMBER() OVER(ORDER BY id) rn FROM emp e ) WHERE rn BETWEEN 21 AND 30;
UNION / INTERSECT / MINUS
集合运算：并集 / 交集 / 差集
SELECT name FROM emp UNION ALL SELECT name FROM contractor; SELECT dept_id FROM emp INTERSECT SELECT id FROM dept;
分析函数（窗口函数）
ROW_NUMBER / RANK / LAG / LEAD
SELECT name, dept, sal, RANK() OVER( PARTITION BY dept ORDER BY sal DESC ) AS rank_in_dept, LAG(sal) OVER(ORDER BY id) prev_sal FROM emp;
常用函数
NVL / DECODE / TO_CHAR / SUBSTR / TRUNC 等
NVL(comm, 0) -- NULL替换 DECODE(dept,'IT','技术',...) -- 条件转换 TO_CHAR(SYSDATE,'YYYY-MM-DD') SUBSTR(name, 1, 3) TRUNC(SYSDATE) -- 截断时间 INSTR(name, '张') -- 查找位置

数据字典常用视图速记

  • USER_* — 当前用户拥有的对象（最常用）
  • ALL_* — 当前用户有权限访问的所有对象
  • DBA_* — 数据库中所有对象（需要 DBA 权限）
  • 高频表：user_tables / user_tab_columns / user_indexes / user_constraints / user_source / user_views / user_triggers

DDL（数据定义语言）

用于创建、修改、删除数据库对象（表、索引、视图、序列等）的命令。

CREATE TABLE
创建表
CREATE TABLE emp ( id NUMBER PRIMARY KEY, name VARCHAR2(50) NOT NULL, dept VARCHAR2(20) );
ALTER TABLE
修改表结构（加列 / 改列 / 删列）
ALTER TABLE emp ADD (sal NUMBER(10,2)); ALTER TABLE emp MODIFY (name VARCHAR2(100)); ALTER TABLE emp DROP COLUMN dept;
DROP TABLE
删除表（加 PURGE 跳过回收站）
DROP TABLE emp; DROP TABLE emp PURGE;
TRUNCATE TABLE
快速清空表数据，DDL 操作，不可回滚
TRUNCATE TABLE emp;
CREATE INDEX
创建索引，加速查询
CREATE INDEX idx_emp_name ON emp(name); CREATE UNIQUE INDEX uq_emp_id ON emp(id);
CREATE SEQUENCE
创建自增序列（Oracle 无 AUTO_INCREMENT）
CREATE SEQUENCE seq_emp_id START WITH 1 INCREMENT BY 1;
CREATE VIEW
创建视图，封装复杂查询
CREATE OR REPLACE VIEW v_emp AS SELECT id, name FROM emp WHERE dept='IT';
COMMENT ON
为表 / 列添加注释
COMMENT ON TABLE emp IS '员工表'; COMMENT ON COLUMN emp.id IS '员工ID';

DML（数据操作语言）— 增删改

INSERT / UPDATE / DELETE 及批量操作命令。（SELECT 查询类命令已移至「表结构 & 查询」标签页）

INSERT
插入单行或多行数据
INSERT INTO emp VALUES (1, 'Alice', 'IT'); INSERT INTO emp (id, name) VALUES (2, 'Bob'); INSERT ALL INTO emp VALUES (3, 'Cathy', 'HR') INTO emp VALUES (4, 'David', 'IT') SELECT 1 FROM DUAL;
UPDATE
更新数据行
UPDATE emp SET dept='Finance' WHERE id = 1;
DELETE
删除数据行（DML，可回滚）
DELETE FROM emp WHERE id = 1;
SELECT INTO
查询结果直接赋值给 PL/SQL 变量
SELECT name INTO v_name FROM emp WHERE id = 1;
MERGE INTO
upsert：有则更新，无则插入
MERGE INTO emp e USING src_tab s ON (e.id = s.id) WHEN MATCHED THEN UPDATE SET e.name = s.name WHEN NOT MATCHED THEN INSERT VALUES (s.id, s.name, s.dept);
BULK COLLECT
批量获取数据到集合中
SELECT id BULK COLLECT INTO v_ids FROM emp WHERE dept = 'IT';
FORALL
批量 DML，减少上下文切换，性能大幅提升
FORALL i IN 1..v_ids.COUNT DELETE FROM emp WHERE id = v_ids(i);
EXECUTE IMMEDIATE
动态构建并执行 SQL
EXECUTE IMMEDIATE 'DELETE FROM emp WHERE id = :1' USING v_id;

TCL（事务控制语言）

管理事务的提交、回滚及保存点。

COMMIT
提交当前事务，持久化所有变更
COMMIT; COMMIT WORK; COMMIT COMMENT '批量更新完成';
ROLLBACK
回滚整个事务，撤销所有未提交的变更
ROLLBACK;
SAVEPOINT
在事务中设置回滚标记点
SAVEPOINT before_update;
ROLLBACK TO SAVEPOINT
回滚到指定保存点，保留之前的部分提交
ROLLBACK TO before_update;
SET TRANSACTION
设置事务属性（隔离级别、只读）
SET TRANSACTION READ ONLY; SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;
LOCK TABLE
显式锁定表
LOCK TABLE emp IN EXCLUSIVE MODE NOWAIT;

控制结构

PL/SQL 块内部的条件判断与循环结构。

IF ... ELSIF ... ELSE
条件分支判断
IF v_score >= 90 THEN v_grade := 'A'; ELSIF v_score >= 80 THEN v_grade := 'B'; ELSE v_grade := 'C'; END IF;
CASE WHEN
多分支选择（可用于 SQL 语句）
v_result := CASE WHEN x = 1 THEN 'one' WHEN x = 2 THEN 'two' ELSE 'other' END;
LOOP
基本无限循环，必须搭配 EXIT 退出
LOOP cnt := cnt + 1; EXIT WHEN cnt > 10; END LOOP;
WHILE LOOP
前置条件循环
WHILE cnt <= 10 LOOP cnt := cnt + 1; END LOOP;
FOR LOOP
计数循环（REVERSE 倒序）
FOR i IN 1..10 LOOP DBMS_OUTPUT.PUT_LINE(i); END LOOP; FOR i IN REVERSE 10..1 LOOP ...
EXIT / CONTINUE
跳出循环 / 跳过当前迭代
EXIT WHEN done; CONTINUE WHEN skip;
GOTO
无条件跳转（谨慎使用）
GOTO cleanup; <<cleanup>> NULL;
NULL
空语句占位，常用于异常处理中
IF x IS NULL THEN NULL; -- 什么都不做 END IF;

游标（Cursor）

逐行处理查询结果集的机制，包含显式游标与隐式游标。

显式游标声明
带参数的游标声明
CURSOR c_emp(p_dept VARCHAR2) IS SELECT id, name FROM emp WHERE dept = p_dept;
OPEN / FETCH / CLOSE
显式管理游标生命周期
OPEN c_emp('IT'); LOOP FETCH c_emp INTO v_id, v_name; EXIT WHEN c_emp%NOTFOUND; ... END LOOP; CLOSE c_emp;
FOR 游标循环（推荐）
自动 OPEN / FETCH / CLOSE
FOR rec IN (SELECT * FROM emp WHERE dept='IT') LOOP DBMS_OUTPUT.PUT_LINE(rec.name); END LOOP;
REF CURSOR
动态 / 可传递的游标引用
TYPE t_cur IS REF CURSOR; v_cur t_cur; OPEN v_cur FOR 'SELECT * FROM emp WHERE dept = :1' USING 'IT'; FETCH v_cur INTO v_rec;
%ROWTYPE
声明与表行结构一致的记录变量
v_emp emp%ROWTYPE; v_emp.name := 'Alice';
%TYPE
声明与列类型一致的变量
v_name emp.name%TYPE; v_sal emp.sal%TYPE;

游标属性速记

  • %FOUND — 最近一次 FETCH 取到数据返回 TRUE
  • %NOTFOUND — 与 FOUND 相反，常用于循环退出
  • %ROWCOUNT — 当前已 FETCH 的行数
  • %ISOPEN — 游标是否已打开

异常处理

捕获、处理或抛出异常，保证程序健壮性。

EXCEPTION 块
异常处理区域
BEGIN ... EXCEPTION WHEN NO_DATA_FOUND THEN DBMS_OUTPUT.PUT_LINE('未找到数据'); WHEN OTHERS THEN RAISE; END;
RAISE
重新抛出当前异常
WHEN OTHERS THEN ROLLBACK; RAISE;
RAISE_APPLICATION_ERROR
主动抛出带自定义错误码和信息
RAISE_APPLICATION_ERROR( -20001, '用户余额不足，当前余额: ' || v_bal );
自定义异常
声明并使用自己的异常变量
e_invalid EXCEPTION; PRAGMA EXCEPTION_INIT(e_invalid, -2290); RAISE e_invalid;
SQLCODE / SQLERRM
获取错误码和错误信息文本
DBMS_OUTPUT.PUT_LINE( 'Error: ' || SQLCODE || ' - ' || SQLERRM );
常见内置异常
无需声明的预定义异常
NO_DATA_FOUND -- 单行查询无结果 TOO_MANY_ROWS -- 单行查询返回多行 DUP_VAL_ON_INDEX -- 唯一索引冲突 ZERO_DIVIDE -- 除零错误 VALUE_ERROR -- 类型转换错误 INVALID_NUMBER -- 非法数字 LOGIN_DENIED -- 登录拒绝 PROGRAM_ERROR -- 内部错误 STORAGE_ERROR -- 存储不足 TIMEOUT_ON_RESOURCE -- 资源超时

存储过程 & 函数 & 包

封装业务逻辑到可复用的数据库对象中。

CREATE PROCEDURE
创建存储过程（无返回值）
CREATE OR REPLACE PROCEDURE p_add_emp( p_id IN NUMBER, p_name IN VARCHAR2, p_out OUT NUMBER ) IS BEGIN INSERT INTO emp VALUES (p_id, p_name); p_out := SQL%ROWCOUNT; END;
CREATE FUNCTION
创建函数（必须 RETURN 值）
CREATE OR REPLACE FUNCTION f_get_name(p_id NUMBER) RETURN VARCHAR2 IS v_name emp.name%TYPE; BEGIN SELECT name INTO v_name FROM emp WHERE id = p_id; RETURN v_name; END;
CREATE TRIGGER
创建触发器（BEFORE / AFTER / INSTEAD OF）
CREATE OR REPLACE TRIGGER trg_emp_audit AFTER INSERT OR UPDATE ON emp FOR EACH ROW BEGIN INSERT INTO audit_log VALUES (:OLD.id, :NEW.name, SYSDATE); END;
CREATE PACKAGE
包规范（接口声明）
CREATE OR REPLACE PACKAGE pkg_emp IS PROCEDURE add_emp(p_id NUMBER, p_name VARCHAR2); FUNCTION get_count RETURN NUMBER; END pkg_emp;
CREATE PACKAGE BODY
包体实现（具体逻辑）
CREATE OR REPLACE PACKAGE BODY pkg_emp IS PROCEDURE add_emp(p_id NUMBER, p_name VARCHAR2) IS BEGIN INSERT INTO emp VALUES (p_id, p_name); END; FUNCTION get_count RETURN NUMBER IS v_cnt NUMBER; BEGIN SELECT COUNT(*) INTO v_cnt FROM emp; RETURN v_cnt; END; END pkg_emp;
调用示例
存储过程和函数的调用方式
-- 匿名块调用 BEGIN p_add(1, 'Alice'); DBMS_OUTPUT.PUT_LINE(f_get(1)); END; -- SQL 中只能调用函数 SELECT f_get_name(id) FROM emp; -- CALL 语句（过程） CALL p_add_emp(2, 'Bob'); -- EXEC 简写（SQL*Plus） EXEC p_add_emp(3, 'Cathy');

完整匿名块模板

DECLARE
  -- 变量、游标、异常声明区
  v_name   emp.name%TYPE;
  v_cnt    NUMBER := 0;
  v_sql    VARCHAR2(2000);

BEGIN
  -- 执行区：SQL / 条件 / 循环 / 调用
  SELECT name INTO v_name FROM emp WHERE id = 1;

  IF v_name IS NOT NULL THEN
    UPDATE emp SET dept = 'IT' WHERE name = v_name;
    v_cnt := SQL%ROWCOUNT;
  END IF;

  COMMIT;
  DBMS_OUTPUT.PUT_LINE('更新行数: ' || v_cnt);

EXCEPTION
  WHEN NO_DATA_FOUND THEN
    DBMS_OUTPUT.PUT_LINE('未找到员工');
  WHEN OTHERS THEN
    ROLLBACK;
    RAISE;
END;
